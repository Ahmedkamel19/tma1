import java.io.*;
public class Laundry {
private ArrayList machines;
public Laundry() {
this.machines = new ArrayList();
}
public void addDataFromFile(File file){
String brandName;
int productionYear;
int capacity;
try{
Scanner read = new Scanner(file);
while (read.hasNextLine()){
String lines = read.nextLine();
String[] slicing = lines.split(" ");
brandName = slicing[0];
productionYear = Integer.parseInt(slicing[1]);
capacity = Integer.parseInt(slicing[2]);
WashingMachine obj = new
WashingMachine(brandName,productionYear,capacity);
if (!(this.machines.contains(obj))){
this.machines.add(obj);
}else{
System.out.println("Not added, Machine exists already: "+obj);
}
}
}catch(Exception e){
System.out.println(e);
}
}
public int brandFrequency(String brandName){
int counter = 0 ;
for (WashingMachine m: machines) {
if (m.getBrandName().equals(brandName)){
counter++;
}
}
return counter;
}

Collections.sort(machines);
